C Resources

Language Snapshot

Overview

  • Language Type: General-purpose, procedural
  • Key Features:
    • Procedural Language: Emphasizes procedures and functions for structured programming
    • Low-Level Access: Supports direct manipulation of memory using pointers
    • Portable: Programs written in C can be easily moved across different platforms
    • Fast and Efficient: Ideal for applications that require direct hardware interaction and optimized performance
    • Rich Library Support: Extensive standard library functions for input/output, string manipulation, math, and more

Invention

  • Creator: Dennis Ritchie
  • Year Created: 1972
  • Major Influence: System programming, particularly the development of Unix
  • Influenced Languages: C++, C#, Java, Python

Notable Frameworks and Libraries

  • Operating Systems: Unix, Linux
  • Compilers: GCC, Clang

Common Use Cases

  • System Programming: Writing operating systems and drivers
  • Embedded Systems: Programming hardware with limited resources
  • Performance-Critical Applications: Applications that require optimized performance
  • Game Development: Low-level programming for high-performance game engines

Demonstrates how a switch statement works to execute different code based on the value of a variable.

                #include <stdio.h>

6. Recursion

Recursion is a process in which a function calls itself to solve smaller instances of the same problem.

            #include 

            int factorial(int n) {
                if (n == 0) {
                    return 1;
                } else {
                    return n * factorial(n - 1);
                }
            }

            int main() {

                return 0;
            }

Expert Level

1. Advanced Pointers and Pointer Arithmetic

Pointer arithmetic allows manipulation of pointers to traverse through arrays or memory blocks.

              #include <stdio.h>
              
              int main() {
                  int arr[] = {10, 20, 30, 40, 50};
                  int *ptr = arr;
                  for (int i = 0; i < 5; i++) {
                      printf("Value: %d, Address: %p\n", *(ptr + i), (ptr + i));
                  }
                  return 0;
              }

2. Function Pointers

Function pointers point to executable code within a function, enabling dynamic function calls.

              #include <stdio.h>
              
              void printMessage() {
                  printf("Hello from a function pointer!\n");
              }
              
              int main() {
                  void (*funcPtr)() = printMessage;
                  funcPtr();  // Calling function using pointer
                  return 0;
              }

3. Double Pointers

Double pointers store the address of another pointer, allowing indirect memory access.

              #include <stdio.h>
              
              int main() {
                  int a = 10;
                  int *ptr = &a;
                  int **dPtr = &ptr;
              
                  printf("Value of a: %d\n", a);
                  printf("Value using single pointer: %d\n", *ptr);
                  printf("Value using double pointer: %d\n", **dPtr);
                  return 0;
              }

4. Structures with Pointers

5. Linked List Implementation

A linked list is a dynamic data structure where each element points to the next one.

              #include <stdio.h>
              #include <stdlib.h>
              
              struct Node {
                  int data;
                  struct Node* next;
              };
              
              void printList(struct Node* node) {
                  while (node != NULL) {
                      printf("%d -> ", node->data);
                      node = node->next;
                  }
                  printf("NULL\n");
              }
              
              int main() {
                  struct Node* head = NULL;
                  struct Node* second = NULL;
                  struct Node* third = NULL;
              
                  head = (struct Node*)malloc(sizeof(struct Node));
                  second = (struct Node*)malloc(sizeof(struct Node));
                  third = (struct Node*)malloc(sizeof(struct Node));
              
                  head->data = 1; head->next = second;
                  second->data = 2; second->next = third;
                  third-> ;data = 3; third->next = NULL;
              
                  printList(head);
              
                  return 0;
              }

6. Dynamic 2D Arrays (Using Pointers)

Dynamically allocates memory for a 2D array using pointers and loops through its elements.

              #include <stdio.h>
              #include <stdlib.h>
              
              int main() {
                  int rows = 3, cols = 3;
                  int **arr = (int **)malloc(rows * sizeof(int *));
                  for (int i = 0; i < rows; i++) {
                      arr[i] = (int *)malloc(cols * sizeof(int));
                  }
              
                  for (int i = 0; i < rows; i++) {
                      for (int j = 0; j < cols; j++) {
                          arr[i][j] = i + j;
                      }
                  }
              
                  for (int i = 0; i < rows; i++) {
                      for (int j = 0; j < cols; j++) {
                          printf("%d ", arr[i][j]);
                      }
                      printf("\n");
                  }
              
                  for (int i = 0; i < rows; i++) {
                      free(arr[i]);
                  }
                  free(arr);
              
                  return 0;
              }

7. Bitwise Operators in C

Bitwise operators perform bit-level operations on data, often used in low-level programming.

              #include <stdio.h>
              
              int main() {
                  int a = 5;  // 0101 in binary
                  int b = 9;  // 1001 in binary
                  
                  printf("a & b = %d\n", a & b);  // AND operation
                  printf("a | b = %d\n", a | b);  // OR operation
                  printf("a ^ b = %d\n", a ^ b);  // XOR operation
                  printf("~a = %d\n", ~a);        // NOT operation
                  return 0;
              }

8. Multi-Threading in C (POSIX Threads)

Multi-threading enables the execution of multiple threads simultaneously.

              #include <stdio.h>
              #include <pthread.h>
              
              void *printMessage(void *arg) {
                  printf("Thread message: %s\n", (char *)arg);
                  return NULL;
              }
              
              int main() {
                  pthread_t thread1, thread2;
                  pthread_create(&thread1, NULL, printMessage, "Hello from Thread 1");
                  pthread_create(&thread2, NULL, printMessage, "Hello from Thread 2");
              
                  pthread_join(thread1, NULL);
                  pthread_join(thread2, NULL);
              
                  return 0;
              }

Reverse a String

Problem: Reverse a string using pointers.

              #include <stdio.h>
              #include <string.h>
              
              void reverseString(char *str) {
                  int n = strlen(str);
                  for (int i = 0; i < n / 2; i++) {
                      char temp = str[i];
                      str[i] = str[n - i - 1];
                      str[n - i - 1] = temp;
                  }
              }
              
              int main() {
                  char str[] = "Hello";
                  reverseString(str);
                  printf("Reversed String: %s\n", str);
                  return 0;
              }

Expert Level

Check Armstrong Number

Problem: Check if a number is an Armstrong number.

              #include <stdio.h>
              #include <math.h>
              
              int isArmstrong(int num) {
                  int original = num, sum = 0, digits = 0;
                  while (original != 0) {
                      original /= 10;
                      digits++;
                  }
                  original = num;
                  while (original != 0) {
                      sum += pow(original % 10, digits);
                      original /= 10;
                  }
                  return sum == num;
              }
              
              int main() {
                  int num = 153;
                  if (isArmstrong(num))
                      printf("%d is an Armstrong number.\n", num);
                  else
                      printf("%d is not an Armstrong number.\n", num);
                  return 0;
              }
Matrix Multiplication

Problem: Multiply two matrices.

              #include <stdio.h>
              
              void multiplyMatrices(int firstMatrix[][2], int secondMatrix[][2], int result[][2], int row1, int row2) {
                  for (int i = 0; i < row1; i++) {
                      for (int j = 0; j < row2; j++) {
                          result[i][j] = 0;
                          for (int k = 0; k < row2; k++) {
                              result[i][j] += firstMatrix[i][k] * secondMatrix[k][j];
                          }
                      }
                  }
              }
              
              int main() {
                  int firstMatrix[2][2] = {{1, 2}, {3, 4}};
                  int secondMatrix[2][2] = {{1, 2}, {3, 4}};
                  int result[2][2];
              
                  multiplyMatrices(firstMatrix, secondMatrix, result, 2, 2);
              
                  printf("Result matrix:\n");
                  for (int i = 0; i < 2; i++) {
                      for (int j = 0; j < 2; j++) {
                          printf("%d ", result[i][j]);
                      }
                      printf("\n");
                  }
                  return 0;
              }
